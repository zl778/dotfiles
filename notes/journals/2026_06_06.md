在alfredr的workflows里，新增 inport keyword 名为voc；
在actions下增加Run Script 内容如下：
```
WORD_FILE="/Users/liangzhu/Library/Mobile Documents/iCloud~md~obsidian/Documents/PKM/pages/单词收集器.md"

mkdir -p "$(dirname "$WORD_FILE")"

WORD="$1"

ENCODED_WORD=$(python3 -c 'import urllib.parse,sys; print(urllib.parse.quote(sys.argv[1]))' "$WORD")

TRANSLATION=$(curl -s "https://api.mymemory.translated.net/get?q=$ENCODED_WORD&langpair=en|zh-CN" | python3 -c 'import json,sys; print(json.load(sys.stdin)["responseData"]["translatedText"])')

echo "- $(date '+%Y-%m-%d') $WORD - $TRANSLATION" >> "$WORD_FILE"
echo "Saved: $WORD - $TRANSLATION"
```
